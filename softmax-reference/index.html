    def one_hot(targets):
    """
    One-hot encoding
    See [source] for info
    Author: Hanan Gharayba, 2023
    """
    targets -= targets.min()
    categories = int(targets.max() + 1)

    oh = np.zeros((len(targets), categories), dtype = int)

    for i in range(len(targets)):
        oh[i, int(targets[i])] = 1

    return oh
